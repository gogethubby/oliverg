      position: absolute;

  <!-- 🐦 Bird Image (Replace src if needed) -->
  <img id="bird" src="https://upload.wikimedia.org/wikipedia/commons/thumb/6/6e/Toucan_icon.png/512px-Toucan_icon.png" alt="Bird" />

  <!-- GSAP Core -->
  <script src="https://cdnjs.cloudflare.com/ajax/libs/gsap/3.12.2/gsap.min.js"></script>

  <script>
    // Bird bobbing up/down = flapping illusion
    gsap.to("#bird", {
      y: -20,
      duration: 0.2,
      repeat: -1,
      yoyo: true,
      ease: "power1.inOut"
    });

    // Bird moving along x-axis
    gsap.to("#bird", {
      x: window.innerWidth + 100,  // fly across screen
      duration: 10,
      repeat: -1,
      ease: "none",
      onRepeat: function() {
        gsap.set("#bird", { x: -100 }); // reset to start
      }
    });
  </script>

</body>
